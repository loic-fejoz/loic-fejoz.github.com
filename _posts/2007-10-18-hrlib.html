Exemples en Python

import Heredis
import sys
#ouvre le fichier passé en paramètre
hf = Heredis.openHeredisFile(sys.argv[1])
#écrit le dictionnaire des noms
for name in hf.getTable('TH5TableDicoNoms'):
    print name.getId(),'\t',name.name
#on ferme le fichier
hf.close
